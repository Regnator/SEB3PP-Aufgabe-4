		log.finest("Das Ergebnis ist " + ergebniss);

	}

	public boolean isEven(int zahl) {
		if( (zahl&1) == 0){ 
			System.out.printf(" Die eingegebene Zahl ist gerade " , zahl); 
			return true;
        }else{ 
        	System.out.printf(" Die eingegebene Zahl ist ungerade ", zahl);
